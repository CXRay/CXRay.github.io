typedef struct linkStack{
    int num;
    struct linkStack *next;
} *linkStackPointer;

linkStackPointer stack = NULL;

bool isFull()
{
    return false;
}

bool isEmpty()
{
    if(stack != NULL)
        return false;
    else
        return true;
}

void push(int n)
{
    linkStackPointer point = (linkStackPointer)malloc(sizeof(struct linkStack));
    point->num = n;
    point->next = stack;
    stack = point;
}

int pop()
{
    if (stack != NULL)
    {
        int n = stack->num;
        linkStackPointer tmp = (linkStackPointer)malloc(sizeof(struct linkStack));
        tmp = stack;
        stack = stack->next;
        free(tmp);
        return n;
    }
    else
        return NULL;
}



int main()
{
    int num;
    cin>>num;
    while (num != 0)
    {
        push(num % 2);
        num = num / 2;
    }
    while (!isEmpty())
    {
        cout<<pop();
    }

    return 0;
}
